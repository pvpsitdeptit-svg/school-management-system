# Firebase Authentication Configuration
# Add these to your .env file

FIREBASE_API_KEY=your_firebase_api_key_here
FIREBASE_PROJECT_ID=your_firebase_project_id_here
FIREBASE_AUTH_DOMAIN=your_project.firebaseapp.com
FIREBASE_DATABASE_URL=https://your_project.firebaseio.com
FIREBASE_STORAGE_BUCKET=your_project.appspot.com
FIREBASE_MESSAGING_SENDER_ID=123456789012
FIREBASE_APP_ID=1:123456789012:web:abcdef123456789012345678

# How to get these values:
# 1. Go to Firebase Console: https://console.firebase.google.com/
# 2. Select or create a project
# 3. Go to Project Settings (gear icon)
# 4. Under "Your apps", select Web App
# 5. Copy the Firebase configuration
# 6. The API key is in the config or under "Service accounts" -> "Generate new private key"
# 7. The Project ID is in the Project Settings general tab
